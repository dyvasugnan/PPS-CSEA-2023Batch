void insertion_sort(int arr[],int n){
    int i,j;
    for(i=1;i<n;i++){
        int key = arr[i];
        j=i-1;
        while(key<arr[i] && j>=0){
            arr[j+1] = arr[j];
            j--;
        }
        arr[j+1] = key;
    }
}
